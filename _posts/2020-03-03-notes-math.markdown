Log
---
$$
\begin{split}
\log_{2}y &= x    \\
        y &= 2^x  \\
\end{split}
$$

let x = 4
$$
\begin{split}
\log_{2}y &= 4    \\
        y &= 2^4  \\
        y &= 16
\end{split}
$$

let y = 16
$$
\begin{split}
\log_{2}16 &= x    \\
        4  &= x     \\
\end{split}
$$

let log base = 10, x = 4
$$
\begin{split}
\log_{10}y &= 4    \\
         y &= 10^4  \\
         y &= 10000
\end{split}
$$

&x   &= \sqrt[2]{y} \\
&x^2 &= y
